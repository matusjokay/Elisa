|   |-- elisa-client-new           # klientská časť

|   |   |-- tsconfig.json          # metadáta kompilátora
|   |   |-- README.md              # dokumentácia klientskej časti
|   |   |-- src                    # zdrojový kód klienta
|   |   |   |-- @elisa             # doplnkové súbory a  štýly
